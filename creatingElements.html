Creating new list items ( li)

            
            const menu = document.querySelector('#menu')
            
            let li = document.createElement('li')
            li.textContent = 'Products'
            menu.appendChild(li)

            let li2 = document.createElement('li')
            li2.textContent = 'About us'
            menu.appendChild(li2)
